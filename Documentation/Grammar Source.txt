file:
    statements? END

statements:
    <empty>
    statements statement NEWLINE

statement:
    assignment
    symbol
    section

assignment:
    visibility? NAME = expression NEWLINE

symbol:
    visibility NAME symbol-parameter*
    visibility NAME symbol-parameter* { NEWLINE symbol-body }

section:
    .SECTION NAME

symbol-body:
    <empty>
    symbol-body symbol-body-line NEWLINE

visibility:
    .GLOBAL
    .LOCAL

symbol-body-line:
    assignment
    label? instruction
    label

label:
    visibility? NAME :

instruction:
    NAME instruction_arguments? NEWLINE
    instruction_arguments NEWLINE

instruction_arguments:
    instruction_argument
    instruction_arguments instruction_argument

instruction_argument:
    PUNCTUATION
    KEYWORD
    expression
