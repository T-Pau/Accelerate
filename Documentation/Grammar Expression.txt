expression:
    multiplicative_term
    expression + multiplicative_term
    expression - multiplicative_term
    expression | multiplicative_term

multiplicative_term:
    unary_term
    multiplicative_term * unary_term
    multiplicative_term / unary_term
    ; multiplicative_term mod unary_term
    multiplicative_term & unary_term
    multiplicative_term ^ unary_term
    multiplicative_term << unary_term
    multiplicative_term >> unary_term

unary_term:
    operand
    - operand
    + operand
    ^ operand
    < operand
    > operand

operand:
    INTEGER
    NAME
    ( expression )
