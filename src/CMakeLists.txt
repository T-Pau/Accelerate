        AddressingMode.cc

        Assembler.cc
        CPU.cc
        CPUParser.cc
        Commandline.cc
        Environment.cc
        Exception.cc
        ExpressionList.cc

        ExpressionParser.cc
        FileParser.cc
        FileReader.cc

        Int.cc

        MemoryMap.cc

        ParseException.cc
        ParsedValue.cc

        Section.cc
        SymbolTable.cc
        Token.cc
        TokenGroup.cc
        TokenNode.cc
        Tokenizer.cc
        TokenizerFile.cc
        TokenizerSequence.cc
        Util.cc
        ${COMPATIBILITY}
)

target_include_directories(xlr8 PRIVATE ${PROJECT_BINARY_DIR})

foreach(PROGRAM xlr8-as xlr8-ld)
    ADD_EXECUTABLE(${PROGRAM} ${PROGRAM}.cc)
    target_include_directories(${PROGRAM} PRIVATE ${PROJECT_BINARY_DIR})
    target_link_libraries(${PROGRAM} xlr8)
    INSTALL(TARGETS ${PROGRAM} RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR})
endforeach()
